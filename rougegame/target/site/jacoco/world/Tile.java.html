/**
 *
 * @author Aeranythe Echosong
 */
public enum Tile {

    FLOOR((char)13),

    WALL((char)12),

    BOUNDS((char)123);

    private char glyph;

    public char glyph() {
        return glyph;
    }

    private Color color;

    public Color color() {
        return color;
    }

    public boolean isDiggable() {
        return this != Tile.WALL;
    }

    public boolean isGround() {
        return this != Tile.WALL && this != Tile.BOUNDS;
    }

    Tile(char glyph) {
        this.glyph = glyph;
    }
}
